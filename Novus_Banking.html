Project Overview

Business Challenge & Solution Approach

Novus Banking required a comprehensive Salesforce solution to modernize their Fixed Deposit application process and streamline investment tracking workflows. As the lead Salesforce developer, I architected and implemented custom Lightning Web Components to support FD Details and Applicant Details objects, while establishing a robust Interest Scheme master object to manage complex investment products. The project focused on eliminating manual data entry processes, reducing operational errors, and providing real-time visibility into customer investment portfolios.

The solution enhanced customer experience through faster application processing and improved operational efficiency for banking staff.

fetchSchemeHandler(){
    let isValid=true;
    let inputFields=this.template.querySelectorAll('.clsFetchScheme');
    console.log(inputFields);
    inputFields.forEach(e=>{
        if(!e.checkValidity()){
            e.checkValidity();
            e.reportValidity();
            isValid=false;
        }
    })
    if(isValid){
        fetchIntScheme({
            depType: this.depositTypeValue, 
            tenMonths: this.tenorInMonthsValue, 
            tenDays: this.tenorInDaysValue, 
            fdAmount: this.fdAmountValue,
            cusType: this.customerTypeValue
        })
            .then(result => {
               console.log(result);
               let options=[];
               result.forEach(e=>{
                let temObj={};
                temObj.label=e.Name;
                temObj.value=e.Id;
                temObj.intRate=e.Interest_Rate__c; 
                options.push(temObj);
               })
               this.schemeOptions=options;
            })
            .catch(error => {
                console.log(error);
            }); 
    }
} 
schemeHandler(event){
    let sch=event.target.value;
    this.schemeValue=event.target.value;
    this.schemeOptions.forEach(scheme=>{
        if(scheme.value==sch){
            this.schemeValue=scheme.value;
            this.interestRateValue=scheme.intRate;
        }
    })
} 
saveHandler(){
    let isValid=true;
    let inputFields=this.templatequerySelectorAll('.clsFetchScheme');
    inputFields.forEach(e=>{
        if(!e.checkValidity()){
            e.checkValidity();
            e.reportValidity();
            isValid=false;
        }
    })
    isValid=true;
    inputFields=this.template.querySelectorAll('.clsSave');
    inputFields.forEach(e=>{
if(!e.checkValidity()){
    e.reportValidity();
    isValid=false;
}
    })
    if(isValid){ 
        updateFd({
            fdId:this.recordId,
            cusType:this.customerTypeValue,
            depType:this.depositTypeValue,
            tenMonths:this.tenorInMonthsValue,
            tenDays:this.tenorInDaysValue,
            payFreq:this.payoutFreqValue,
            fdAmount:this.fdAmountValue,
            insSchemeId:this.schemeValue,
            intRate:this.interestRateValue
        }) 
            .then(result => {
                console.log(result);
                this.dispatchEvent(new ShowToastEvent({
                    title: "Success!",
                    message: "Successfully Record is Updated!",
                    variant: "success"
                }));
            })
            .catch(error => {
                console.log(error);
                this.dispatchEvent(new ShowToastEvent({
                    title: "Error!",
                    message: "We could not update fd record, check error!",
                    variant: "Error"
                }));
            })
    }
}
